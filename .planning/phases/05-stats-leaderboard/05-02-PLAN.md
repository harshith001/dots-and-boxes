const BASE = process.env.NEXT_PUBLIC_BACKEND_URL ?? 'http://localhost:3001';

export async function postSession(username: string): Promise<{ username: string }> {
  const res = await fetch(`${BASE}/api/session`, {
    method: 'POST',
    headers: { 'Content-Type': 'application/json' },
    credentials: 'include',
    body: JSON.stringify({ username }),
  });
  if (!res.ok) throw new Error('Failed to set session');
  return res.json() as Promise<{ username: string }>;
}

export async function getSession(): Promise<{ username: string | null }> {
  const res = await fetch(`${BASE}/api/session`, { credentials: 'include' });
  if (res.status === 404) return { username: null };
  return res.json() as Promise<{ username: string | null }>;
}

export interface PlayerStats {
  username: string;
  totalMatches: number;
  wins: number;
  losses: number;
  draws: number;
  winRate: number;
}

export interface MatchRecord {
  id: number;
  opponentName: string;
  result: 'win' | 'loss' | 'draw';
  myScore: number;
  opponentScore: number;
  playedAt: number;
}

export interface LeaderboardEntry {
  rank: number;
  username: string;
  wins: number;
  totalMatches: number;
}

export async function getStats(username: string): Promise<{ stats: PlayerStats; history: MatchRecord[] }> {
  const res = await fetch(`${BASE}/api/stats/${encodeURIComponent(username)}`, { credentials: 'include' });
  if (!res.ok) throw new Error('Failed to fetch stats');
  return res.json() as Promise<{ stats: PlayerStats; history: MatchRecord[] }>;
}

export async function getLeaderboard(): Promise<{ leaderboard: LeaderboardEntry[] }> {
  const res = await fetch(`${BASE}/api/leaderboard`, { credentials: 'include' });
  if (!res.ok) throw new Error('Failed to fetch leaderboard');
  return res.json() as Promise<{ leaderboard: LeaderboardEntry[] }>;
}

'use client';

import { useEffect, useState } from 'react';
import { useRouter } from 'next/navigation';
import { getStats, getSession } from '@/lib/api';
import type { PlayerStats, MatchRecord } from '@/lib/api';

  const router = useRouter();
  const [username, setUsername] = useState<string>('');
  const [stats, setStats] = useState<PlayerStats | null>(null);
  const [history, setHistory] = useState<MatchRecord[]>([]);
  const [loading, setLoading] = useState(true);

  useEffect(() => {
    async function load() {
      try {
        const session = await getSession();
        const name = session.username ?? sessionStorage.getItem('operatorName') ?? '';
        if (!name) { router.push('/'); return; }
        setUsername(name);
        const data = await getStats(name);
        setStats(data.stats);
        setHistory(data.history);
      } catch {
        // no-op
      } finally {
        setLoading(false);
      }
    }
    void load();
  }, [router]);

  return (
    <div className="min-h-screen bg-background text-primary font-body overflow-hidden">
      {/* TopAppBar */}
      <header className="fixed top-0 w-full z-50 flex justify-between items-center px-6 h-14 bg-surface border-b border-surface-variant/20">
        <span className="font-headline font-bold tracking-tighter text-lg">KINETIC_GRID</span>

          OPERATOR: <span className="text-primary-fixed">{username}</span>
        </span>

      <nav className="fixed left-0 top-14 h-[calc(100vh-3.5rem)] hidden lg:flex flex-col items-center py-8 z-40 bg-surface w-20 border-r border-surface-variant/20">
        {[
          { icon: 'grid_4x4', label: 'GRID', path: '/lobby' },

            }`}

'use client';

import { useEffect, useState, useCallback } from 'react';
import { useRouter } from 'next/navigation';
import { getLeaderboard } from '@/lib/api';
import type { LeaderboardEntry } from '@/lib/api';

const POLL_INTERVAL = 30_000;

export default function LeaderboardPage() {
  const router = useRouter();
  const [entries, setEntries] = useState<LeaderboardEntry[]>([]);
  const [loading, setLoading] = useState(true);
  const [lastUpdated, setLastUpdated] = useState<Date | null>(null);

  const fetchLeaderboard = useCallback(async () => {
    try {
      const data = await getLeaderboard();
      setEntries(data.leaderboard);
      setLastUpdated(new Date());
    } catch { /* no-op */ } finally {
      setLoading(false);
    }
  }, []);

  useEffect(() => {
    void fetchLeaderboard();
    const interval = setInterval(() => { void fetchLeaderboard(); }, POLL_INTERVAL);
    return () => clearInterval(interval);
  }, [fetchLeaderboard]);

  return (
    <div className="min-h-screen bg-background text-primary font-body overflow-hidden">
      {/* TopAppBar */}
      <header className="fixed top-0 w-full z-50 flex justify-between items-center px-6 h-14 bg-surface border-b border-surface-variant/20">
        <span className="font-headline font-bold tracking-tighter text-lg">KINETIC_GRID</span>
        <div className="flex items-center gap-4">
          {lastUpdated && (
            <span className="font-label text-[9px] text-secondary/30 uppercase tracking-widest">
              UPDATED: {lastUpdated.toLocaleTimeString('en-US', { hour12: false })}
            </span>
          )}
        </div>
      </header>

      {/* SideNav */}
      <nav className="fixed left-0 top-14 h-[calc(100vh-3.5rem)] hidden lg:flex flex-col items-center py-8 z-40 bg-surface w-20 border-r border-surface-variant/20">
        {[
          { icon: 'grid_4x4', label: 'GRID', path: '/lobby' },
          { icon: 'leaderboard', label: 'LB', path: '/leaderboard', active: true },
          { icon: 'person', label: 'DASH', path: '/dashboard' },
        ].map(({ icon, label, path, active }) => (
          <div
            key={icon}
            onClick={() => router.push(path)}
            className={`flex flex-col items-center gap-1 p-2 cursor-pointer transition-all mb-6 ${
              active ? 'text-primary-fixed border-r-2 border-primary-fixed pr-1' : 'text-secondary/40 hover:text-secondary'
            }`}
          >
            <span className="material-symbols-outlined">{icon}</span>
            <span className="font-label text-[9px] tracking-widest uppercase">{label}</span>
          </div>
        ))}
      </nav>

      <main className="pt-14 lg:pl-20 p-8">
        <div className="max-w-2xl mx-auto space-y-6">
          <div>
            <h1 className="font-headline text-2xl font-bold tracking-tighter">GLOBAL_RANKINGS</h1>
            <p className="font-label text-[10px] uppercase tracking-widest text-secondary/40 mt-1">
              TOP OPERATORS — REFRESHES EVERY 30S
            </p>
          </div>

          {loading ? (
            <div className="font-label text-[10px] tracking-widest text-secondary/40 animate-pulse">LOADING_RANKINGS...</div>
          ) : entries.length === 0 ? (
            <div className="font-label text-[10px] text-secondary/30 tracking-widest">NO_OPERATORS_RANKED_YET</div>
          ) : (
            <div className="space-y-1">
              {/* Header */}
              <div className="flex items-center justify-between px-4 py-2 font-label text-[9px] uppercase tracking-widest text-secondary/30">
                <span className="w-8">RANK</span>
                <span className="flex-1 pl-4">OPERATOR</span>
                <span className="w-16 text-right">VICTORIES</span>
                <span className="w-16 text-right">MATCHES</span>
              </div>

              {entries.map((entry) => (
                <div
                  key={entry.username}
                  className={`flex items-center justify-between px-4 py-3 border transition-colors ${
                    entry.rank <= 3
                      ? 'border-primary-fixed/20 bg-primary-fixed/5'
                      : 'border-outline-variant/10 bg-surface-container hover:border-outline-variant/30'
                  }`}
                >
                  <span className={`font-headline text-sm font-bold w-8 tabular-nums ${
                    entry.rank === 1 ? 'text-primary-fixed' : entry.rank <= 3 ? 'text-primary-fixed/60' : 'text-secondary/40'
                  }`}>
                    #{entry.rank}
                  </span>
                  <span className="flex-1 pl-4 font-headline text-sm tracking-tight">{entry.username}</span>
                  <span className="w-16 text-right font-mono text-sm text-primary-fixed tabular-nums">{entry.wins}</span>
                  <span className="w-16 text-right font-mono text-xs text-secondary/40 tabular-nums">{entry.totalMatches}</span>
                </div>
              ))}
            </div>
          )}
        </div>
      </main>
    </div>
  );
}
