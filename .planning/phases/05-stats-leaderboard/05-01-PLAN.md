```typescript
import Database from 'better-sqlite3';
import path from 'path';
import { fileURLToPath } from 'url';

const __dirname = path.dirname(fileURLToPath(import.meta.url));
const DB_PATH = process.env.DB_PATH ?? path.join(__dirname, '../../data/game.db');

// Ensure data dir exists
import fs from 'fs';
fs.mkdirSync(path.dirname(DB_PATH), { recursive: true });

export const db = new Database(DB_PATH);

// Schema
db.exec(`
  CREATE TABLE IF NOT EXISTS players (
    username TEXT PRIMARY KEY,
    created_at INTEGER NOT NULL DEFAULT (unixepoch())
  );

  CREATE TABLE IF NOT EXISTS matches (
    id INTEGER PRIMARY KEY AUTOINCREMENT,
    player1 TEXT NOT NULL,
    player2 TEXT NOT NULL,
    winner TEXT,
    score_p1 INTEGER NOT NULL,
    score_p2 INTEGER NOT NULL,
    played_at INTEGER NOT NULL DEFAULT (unixepoch()),
    FOREIGN KEY(player1) REFERENCES players(username),
    FOREIGN KEY(player2) REFERENCES players(username)
  );

  CREATE INDEX IF NOT EXISTS idx_matches_player1 ON matches(player1);
  CREATE INDEX IF NOT EXISTS idx_matches_player2 ON matches(player2);
`);

// Query helpers

export interface PlayerStats {
  username: string;
  totalMatches: number;
  wins: number;
  losses: number;
  draws: number;
  winRate: number;
}

export interface MatchRecord {
  id: number;
  opponentName: string;
  result: 'win' | 'loss' | 'draw';
  myScore: number;
  opponentScore: number;
  playedAt: number;
}

export interface LeaderboardEntry {
  rank: number;
  username: string;
  wins: number;
  totalMatches: number;
}

export function upsertPlayer(username: string): void {
  db.prepare('INSERT OR IGNORE INTO players (username) VALUES (?)').run(username);
}

export function recordMatch(
  player1: string,
  player2: string,
  winner: string | null,
  scoreP1: number,
  scoreP2: number,
): void {
  upsertPlayer(player1);
  // Don't upsert bot names as real players
  if (!player2.startsWith('NEXUS_BOT')) upsertPlayer(player2);
  db.prepare(
    'INSERT INTO matches (player1, player2, winner, score_p1, score_p2) VALUES (?, ?, ?, ?, ?)'
  ).run(player1, player2, winner, scoreP1, scoreP2);
}

export function getPlayerStats(username: string): PlayerStats {
  const rows = db.prepare(`
    SELECT
      COUNT(*) as total,
      SUM(CASE WHEN winner = ? THEN 1 ELSE 0 END) as wins,
      SUM(CASE WHEN winner IS NOT NULL AND winner != ? THEN 1 ELSE 0 END) as losses,
      SUM(CASE WHEN winner IS NULL THEN 1 ELSE 0 END) as draws
    FROM matches
    WHERE player1 = ? OR player2 = ?
  `).get(username, username, username, username) as { total: number; wins: number; losses: number; draws: number };

  const total = rows.total ?? 0;
  const wins = rows.wins ?? 0;
  return {
    username,
    totalMatches: total,
    wins,
    losses: rows.losses ?? 0,
    draws: rows.draws ?? 0,
    winRate: total > 0 ? Math.round((wins / total) * 100) : 0,
  };
}

export function getMatchHistory(username: string, limit = 10): MatchRecord[] {
  const rows = db.prepare(`
    SELECT id, player1, player2, winner, score_p1, score_p2, played_at
    FROM matches
    WHERE player1 = ? OR player2 = ?
    ORDER BY played_at DESC
    LIMIT ?
  `).all(username, username, limit) as Array<{
    id: number; player1: string; player2: string;
    winner: string | null; score_p1: number; score_p2: number; played_at: number;
  }>;

  return rows.map(r => {
    const isP1 = r.player1 === username;
    const myScore = isP1 ? r.score_p1 : r.score_p2;
    const oppScore = isP1 ? r.score_p2 : r.score_p1;
    const result = r.winner === null ? 'draw' : r.winner === username ? 'win' : 'loss';
    return {
      id: r.id,
      opponentName: isP1 ? r.player2 : r.player1,
      result,
      myScore,
      opponentScore: oppScore,
      playedAt: r.played_at,
    };
  });
}

// Leaderboard cache (30s)
let leaderboardCache: { data: LeaderboardEntry[]; cachedAt: number } | null = null;

export function getLeaderboard(limit = 20): LeaderboardEntry[] {
  if (leaderboardCache && Date.now() - leaderboardCache.cachedAt < 30_000) {
    return leaderboardCache.data;
  }

  const rows = db.prepare(`
    SELECT
      p.username,
      COUNT(m.id) as total,
      SUM(CASE WHEN m.winner = p.username THEN 1 ELSE 0 END) as wins
    FROM players p
    LEFT JOIN matches m ON m.player1 = p.username OR m.player2 = p.username
    GROUP BY p.username
    ORDER BY wins DESC, total DESC
    LIMIT ?
  `).all(limit) as Array<{ username: string; total: number; wins: number }>;

  const data = rows.map((r, i) => ({
    rank: i + 1,
    username: r.username,
    wins: r.wins ?? 0,
    totalMatches: r.total ?? 0,
  }));

  leaderboardCache = { data, cachedAt: Date.now() };
  return data;
}
```

```typescript
// Session — set/get username via HttpOnly cookie
app.post('/api/session', (req, res) => {
  const { username } = req.body as { username?: string };
  if (!username || typeof username !== 'string' || username.trim().length === 0) {
    res.status(400).json({ error: 'username required' });
    return;
  }
  const name = username.trim().toUpperCase().slice(0, 24);
  upsertPlayer(name);
  res.cookie('operatorName', name, {
    httpOnly: true,
    sameSite: 'lax',
    maxAge: 30 * 24 * 60 * 60 * 1000, // 30 days
  });
  res.json({ username: name });
});

app.get('/api/session', (req, res) => {
  const name = req.cookies?.operatorName;
  if (!name) { res.status(404).json({ username: null }); return; }
  res.json({ username: name });
});

app.delete('/api/session', (_req, res) => {
  res.clearCookie('operatorName');
  res.json({ ok: true });
});

// Stats
app.get('/api/stats/:username', (req, res) => {
  const { username } = req.params;
  const stats = getPlayerStats(username.toUpperCase());
  const history = getMatchHistory(username.toUpperCase(), 10);
  res.json({ stats, history });
});

// Leaderboard
app.get('/api/leaderboard', (_req, res) => {
  res.json({ leaderboard: getLeaderboard(20) });
});
```
