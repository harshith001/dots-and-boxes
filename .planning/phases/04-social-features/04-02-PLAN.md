```tsx
'use client';

import { useState } from 'react';
import type { EmojiReaction } from '@/types/game';

const EMOJIS: EmojiReaction[] = ['😎', '😂', '😡', '🔥', '👏'];
const COOLDOWN_MS = 2500;

interface EmojiPanelProps {
  onSend: (emoji: EmojiReaction) => void;
}

export default function EmojiPanel({ onSend }: EmojiPanelProps) {
  const [cooldownUntil, setCooldownUntil] = useState<number>(0);

  const isCoolingDown = Date.now() < cooldownUntil;

  function handleClick(emoji: EmojiReaction) {
    if (isCoolingDown) return;
    onSend(emoji);
    setCooldownUntil(Date.now() + COOLDOWN_MS);
    // Force re-render after cooldown expires
    setTimeout(() => setCooldownUntil(0), COOLDOWN_MS);
  }

  return (
    <div className={`flex flex-col gap-2 transition-opacity duration-300 ${isCoolingDown ? 'opacity-30' : 'opacity-100'}`}>
      <span className="font-label text-[9px] uppercase tracking-[0.2em] text-secondary/40">REACTIONS</span>
      <div className="flex gap-2">
        {EMOJIS.map(emoji => (
          <button
            key={emoji}
            onClick={() => handleClick(emoji)}
            disabled={isCoolingDown}
            className="w-10 h-10 flex items-center justify-center bg-surface-container border border-outline-variant/20 hover:border-primary-fixed/50 hover:bg-surface-container-high transition-all duration-100 text-lg disabled:cursor-not-allowed active:scale-95"
            title={isCoolingDown ? 'COOLDOWN...' : emoji}
          >
            {emoji}
          </button>
        ))}
      </div>
    </div>
  );
}
```

Create `frontend/components/game/ChatPanel.tsx`:

```tsx
'use client';

import { useState } from 'react';
import type { ChatReceivedEvent } from '@/types/game';

type PredefinedMessage = 'Nice move' | 'Good game' | 'That was close';
const MESSAGES: PredefinedMessage[] = ['Nice move', 'Good game', 'That was close'];
const BUTTON_COOLDOWN_MS = 1500;

interface ChatPanelProps {
  onSend: (message: PredefinedMessage) => void;
  log: ChatReceivedEvent[];
  myRole: 'p1' | 'p2';
  myName: string;
}

export default function ChatPanel({ onSend, log, myRole, myName }: ChatPanelProps) {
  const [cooldowns, setCooldowns] = useState<Record<string, number>>({});

  function handleSend(msg: PredefinedMessage) {
    if (Date.now() < (cooldowns[msg] ?? 0)) return;
    onSend(msg);
    setCooldowns(prev => ({ ...prev, [msg]: Date.now() + BUTTON_COOLDOWN_MS }));
    setTimeout(() => setCooldowns(prev => ({ ...prev, [msg]: 0 })), BUTTON_COOLDOWN_MS);
  }

  return (
    <div className="flex flex-col gap-3">
      <span className="font-label text-[9px] uppercase tracking-[0.2em] text-secondary/40">COMMS</span>

      {/* Chat log */}
      {log.length > 0 && (
        <div className="space-y-1 max-h-24 overflow-y-auto">
          {log.slice(-6).map((entry, i) => (
            <div key={i} className="font-mono text-[10px] leading-snug">
              <span className={entry.fromRole === myRole ? 'text-primary' : 'text-primary-fixed'}>
                {entry.fromRole === myRole ? myName : entry.fromName}
              </span>
              <span className="text-secondary/40 mx-1">›</span>
              <span className="text-secondary/80">{entry.message}</span>
            </div>
          ))}
        </div>
      )}

      {/* Message buttons */}
      <div className="flex flex-col gap-1.5">
        {MESSAGES.map(msg => {
          const isCooling = Date.now() < (cooldowns[msg] ?? 0);
          return (
            <button
              key={msg}
              onClick={() => handleSend(msg)}
              disabled={isCooling}
              className={`px-3 py-2 text-left font-label text-[10px] uppercase tracking-widest border transition-all duration-100 ${
                isCooling
                  ? 'border-outline-variant/10 text-secondary/20 cursor-not-allowed'
                  : 'border-outline-variant/20 text-secondary/60 hover:border-primary-fixed/40 hover:text-primary-fixed'
              }`}
            >
              {msg.toUpperCase().replace(/ /g, '_')}
            </button>
          );
        })}
      </div>
    </div>
  );
}
```
