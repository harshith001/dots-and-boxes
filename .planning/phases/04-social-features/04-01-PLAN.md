Inside `io.on('connection', (socket) => { ... })`, add after the existing handlers:

```typescript
  // Emoji reaction
  socket.on('emoji:send', ({ roomId, playerToken, emoji }: EmojiSendPayload) => {
    const dedupKey = `${roomId}:${playerToken}`;
    const lastSent = emojiLastSent.get(dedupKey) ?? 0;
    if (Date.now() - lastSent < 2000) return; // silently drop within 2s
    emojiLastSent.set(dedupKey, Date.now());

    const room = roomManager.getRoom(roomId);
    if (!room) return;
    const sender = room.players.find(p => p.playerToken === playerToken);
    if (!sender) return;

    // Broadcast to room (both players see it — opponent gets the floating animation)
    io.to(roomId).emit('emoji:received', { emoji, fromRole: sender.role } satisfies EmojiReceivedEvent);
  });

  // Predefined chat
  socket.on('chat:send', ({ roomId, playerToken, message }: ChatSendPayload) => {
    const room = roomManager.getRoom(roomId);
    if (!room) return;
    const sender = room.players.find(p => p.playerToken === playerToken);
    if (!sender) return;

    io.to(roomId).emit('chat:received', {
      message,
      fromRole: sender.role,
      fromName: sender.name,
    } satisfies ChatReceivedEvent);
  });
```

Add the necessary imports at the top of the file (add to existing import line):
