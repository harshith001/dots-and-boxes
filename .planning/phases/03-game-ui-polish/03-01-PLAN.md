```tsx
interface PlayerCardProps {
  name: string;
  score: number;
  role: 'p1' | 'p2';
  isActive: boolean;   // true = this player's turn
  isMe: boolean;       // true = local player
}

export default function PlayerCard({ name, score, role, isActive, isMe }: PlayerCardProps) {
  const accentColor = role === 'p1' ? '#ffffff' : '#c3f400';
  const accentClass = role === 'p1' ? 'text-primary' : 'text-primary-fixed';
  const borderClass = role === 'p1' ? 'border-primary/20' : 'border-primary-fixed/20';
  const activeBorder = role === 'p1' ? 'border-primary/60' : 'border-primary-fixed/60';

  return (
    <div
      className={`flex flex-col items-center gap-4 p-6 border transition-all duration-500 ${
        isActive
          ? `${activeBorder} bg-surface-container/60`
          : `${borderClass} bg-surface-container/20 opacity-40`
      }`}
    >
      {/* Role badge */}
      <div className="flex items-center gap-2">
        {isActive && (
          <span className="relative flex h-1.5 w-1.5">
            <span className={`animate-ping absolute inline-flex h-full w-full rounded-full opacity-75 ${role === 'p1' ? 'bg-primary' : 'bg-primary-fixed'}`} />
            <span className={`relative inline-flex rounded-full h-1.5 w-1.5 ${role === 'p1' ? 'bg-primary' : 'bg-primary-fixed'}`} />
          </span>
        )}
        <span className="font-label text-[9px] uppercase tracking-[0.25em] text-secondary/60">
          {isMe ? 'YOU' : 'OPPONENT'}
        </span>
      </div>

      {/* Score */}
      <div className={`font-headline text-5xl font-bold tabular-nums ${accentClass}`}>
        {score}
      </div>

      {/* Name */}
      <div className="font-headline text-xs tracking-widest uppercase text-secondary truncate max-w-[120px] text-center">
        {name}
      </div>

      {/* Turn label */}
      <div className="font-label text-[9px] uppercase tracking-widest h-3">
        {isActive ? (
          <span className={`${accentClass} animate-pulse`}>▸ ACTIVE</span>
        ) : (
          <span className="text-transparent">_</span>
        )}
      </div>

      {/* Color indicator bar */}
      <div
        className="w-8 h-0.5 transition-all duration-500"
        style={{ backgroundColor: isActive ? accentColor : 'transparent', boxShadow: isActive ? `0 0 8px ${accentColor}66` : 'none' }}
      />
    </div>
  );
}
```
